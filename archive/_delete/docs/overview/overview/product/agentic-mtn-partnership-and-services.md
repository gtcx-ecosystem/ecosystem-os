**AI Credit Scoring Revolution:**

```typescript
interface GTCXCreditScore {
  // Traditional factors (limited weight - 20%)
  traditional: {
    existingCreditHistory?: CreditRecord[]; // Often empty for ASM
    bankAccountHistory?: BankRecord;
  };

  // Verification-based factors (40%)
  verification: {
    gciScore: number; // Compliance score
    verificationConsistency: number; // How regularly they verify
    verificationAge: number; // Length of verification history
    documentCompleteness: number; // % of required docs on file
    auditResults: AuditRecord[]; // Past audit outcomes
  };

  // Transaction-based factors (25%)
  transactions: {
    averageTransactionSize: number;
    transactionFrequency: number;
    paymentPunctuality: number; // Do they pay on time?
    buyerQuality: number; // Quality of buyer relationships
    volumeTrend: 'growing' | 'stable' | 'declining';
  };

  // Social/network factors (15%)
  social: {
    cooperativeMembership: boolean;
    peerEndorsements: Endorsement[];
    networkStrength: number; // Connected verified miners
    communityStanding: CommunityScore;
  };

  // Computed outputs
  outputs: {
    creditScore: number; // 300-850 scale
    maxLoanAmount: number;
    suggestedInterestRate: number;
    riskCategory: 'low' | 'medium' | 'high';
    approvalLikelihood: number;
  };
}
```
